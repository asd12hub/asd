package xenew.Interfaces;


import java.util.Scanner;

//Interface 1
interface CardPayment {
 void processCardPayment(double amount, int pin);
 void refundCardPayment(double amount);
}

//Interface 2
interface UPIPayment {
 void processUPIPayment(double amount, int pin);
 void refundUPIPayment(double amount);
}

//Class implementing both interfaces
class PaymentGateway implements CardPayment, UPIPayment {

 private int correctPin = 1234;
 private double balance = 10000;   // account balance
 private double cardLimit = 5000;  // card transaction limit

 // ---- Card Payment ----
 public void processCardPayment(double amount, int pin) {

     if(pin != correctPin){
         System.out.println("Invalid PIN");
         return;
     }

     if(amount <= 0){
         System.out.println("Amount must be positive");
         return;
     }

     if(amount > cardLimit){
         System.out.println("Exceeds card limit");
         return;
     }

     if(amount > balance){
         System.out.println("Insufficient balance");
         return;
     }

     balance -= amount;
     System.out.println("Card payment successful: " + amount);
 }

 public void refundCardPayment(double amount) {
     balance += amount;
     System.out.println("Card refund done: " + amount);
 }

 // ---- UPI Payment ----
 public void processUPIPayment(double amount, int pin) {

     if(pin != correctPin){
         System.out.println("Invalid PIN");
         return;
     }

     if(amount <= 0){
         System.out.println("Amount must be positive");
         return;
     }

     if(amount > balance){
         System.out.println("Insufficient balance");
         return;
     }

     balance -= amount;
     System.out.println("UPI payment successful: " + amount);
 }

 public void refundUPIPayment(double amount) {
     balance += amount;
     System.out.println("UPI refund done: " + amount);
 }

 // utility
 public void showBalance(){
     System.out.println("Current Balance: " + balance);
 }
 
 
 
 
 // main
 
 public static void main(String[] args) {

     PaymentGateway pg = new PaymentGateway();

     // Card Payment
     pg.processCardPayment(2000, 1234);
     pg.showBalance();

     // UPI Payment
     pg.processUPIPayment(1500, 1234);
     pg.showBalance();

     // Refunds
     pg.refundCardPayment(500);
     pg.refundUPIPayment(300);

     pg.showBalance();
 }
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
}
